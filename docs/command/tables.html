public class ArrayExample {
    public static void main(String[] args) {
        // Δήλωση πίνακα ακεραίων
        int[] integerArray ;
                    
        // Δέσμευση μνήμης για τον πίνακα με 5 θέσεις
        integerArray  = new int[5];
                    
        // Αρχικοποίηση των στοιχείων του πίνακα
        integerArray [0] = 10;
        integerArray [1] = 20;
        integerArray [2] = 30;
        integerArray [3] = 40;
        integerArray [4] = 50;
                    
        // Εκτύπωση των στοιχείων του πίνακα
        for (int i = 0; i < integerArray .length; i++) {
            System.out.println("Στοιχείο στη θέση " + i + ": " + integerArray [i]);
        }
    }
}
